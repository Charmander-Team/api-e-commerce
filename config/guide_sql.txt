CREATE TABLE category (
    id INT PRIMARY KEY NOT NULL,
    name VARCHAR(100),
    image VARCHAR(255),
    `delete` INT
);

CREATE TABLE stock (
    id INT PRIMARY KEY NOT NULL,
    product_id INT,
    quantity INT
);

CREATE TABLE bid (
    id INT PRIMARY KEY NOT NULL,
    product_id INT,
    user_id INT,
    bid_amount INT,
    status VARCHAR(100),
    date DATETIME
);

CREATE TABLE user (
    id INT PRIMARY KEY NOT NULL,
    name VARCHAR(100),
    firstname VARCHAR(100),
    mail VARCHAR(255),
    password VARCHAR(100),
    admin INT
);

CREATE TABLE `order` (
    id INT PRIMARY KEY NOT NULL,
    ref VARCHAR(100),
    user_id INT,
    status VARCHAR(100),
    date DATETIME,
    product_id INT
);

CREATE TABLE invoice (
    id INT PRIMARY KEY NOT NULL,
    user_id INT,
    date DATETIME,
    order_ref VARCHAR(100),
    payment_method VARCHAR(100)
);
